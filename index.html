// ----- server.js ----- const http = require("http"); const express = require("express"); const { Server } = require("socket.io"); const app = express(); const server = http.createServer(app); const io = new Server(server); // Serve the single HTML file app.get("/", (req, res) => { res.send(` Realtime Chat
`); }); // Socket.io handling io.on("connection", (socket) => { console.log("A user connected"); socket.on("chat message", (msg) => { io.emit("chat message", msg); }); socket.on("disconnect", () => { console.log("A user disconnected"); }); }); // Start server const PORT = 3000; server.listen(PORT, () => { console.log("Server running on http://localhost:" + PORT); });
